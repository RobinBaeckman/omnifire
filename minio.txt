setup s3 transition for minio can be done in the pod for bitnami chart:
mc alias set minio http://127.0.0.1:9000 admin123 secret123
curl https://min.io/docs/minio/linux/examples/LifecycleManagementAdmin.json --output LifecycleManagementAdmin.json
mc admin policy add minio LifecycleAdminPolicy LifecycleManagementAdmin.json
mc admin user add minio alphaLifecycleAdmin LongRandomSecretKey
mc admin policy set minio LifecycleAdminPolicy user=alphaLifecycleAdmin
mc ilm tier add s3 minio S3TIER  \
   --access-key AKIAUJ6NIJSHHQVMH3G7      \
   --secret-key 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      \
   --bucket omnifire                 \
	--endpoint https://s3.amazonaws.com \
   --region ap-northeast-1
